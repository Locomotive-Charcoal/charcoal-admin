430:     /**
431:      * @throws Exception If the menu was not properly configured.
432:      * @return array This method is a generator.
433:      */
434:     public function headerMenu()
435:     {
436:         $headerMenu = $this->adminConfig['header_menu'];
437: 
438:         if (!isset($headerMenu['items'])) {
439:             throw new Exception(
440:                 'Header menu was not property configured.'
441:             );
442:         }
443: 
444:         $svgUri = $this->baseUrl().'assets/admin/images/svgs.svg#icon-';
445:         foreach ($headerMenu['items'] as $menuIdent => $menuItem) {
446:             if (isset($menuItem['active']) && $menuItem['active'] === false) {
447:                 continue;
448:             }
449: 
450:             if (!isset($menuItem['ident'])) {
451:                 $menuItem['ident'] = $menuIdent;
452:             }
453: 
454:             if (!empty($menuItem['url'])) {
455:                 $url = $menuItem['url'];
456:                 if ($url && strpos($url, ':') === false && !in_array($url[0], [ '/', '#', '?' ])) {
457:                     $url = $this->adminUrl().$url;
458:                 }
459:             } else {
460:                 $url = '';
461:             }
462:             $menuItem['url'] = $url;
463: 
464:             if (isset($menuItem['icon'])) {
465:                 $icon = $menuItem['icon'];
466:                 if ($icon && strpos($icon, ':') === false && !in_array($icon[0], [ '/', '#', '?' ])) {
467:                     $icon = $svgUri.$icon;
468:                 }
469:             } else {
470:                 $icon = $svgUri.'contents';
471:             }
472: 
473:             if (is_string($icon) && strpos($icon, '.svg') > 0) {
474:                 unset($menuItem['icon']);
475:                 $menuItem['svg'] = $icon;
476:             } else {
477:                 unset($menuItem['svg']);
478:                 $menuItem['icon'] = $icon;
479:             }
480: 
481:             if (isset($menuItem['label']) && TranslationString::isTranslatable($menuItem['label'])) {
482:                 $menuItem['label'] = new TranslationString($menuItem['label']);
483:             }
484: 
485:             $menuItem['show_label'] = (isset($menuItem['show_label']) ? !!$menuItem['show_label'] : true);
486: 
487:             $menuItem['selected'] = ($menuItem['ident'] === filter_input(INPUT_GET, 'main_menu'));
488: 
489:             yield $menuItem;
490:         }
491:     }
