453:     /**
454:      * @return boolean
455:      */
456:     private function authByToken()
457:     {
458:         $authToken = $this->modelFactory()->create('charcoal/admin/object/auth-token');
459: 
460:         if ($authToken->metadata()->enabled() !== true) {
461:             return false;
462:         }
463: 
464:         $cookieName = $authToken->metadata()->cookieName();
465: 
466:         if (!isset($_COOKIE[$cookieName])) {
467:             return false;
468:         }
469: 
470:         $authCookie = $_COOKIE[$cookieName];
471:         $vals = explode(';', $authCookie);
472:         $username = $authToken->getUsernameFromToken($vals[0], $vals[1]);
473:         if (!$username) {
474:             return false;
475:         }
476: 
477:         $u = $this->modelFactory()->create('charcoal/admin/user');
478: 
479:         $u->load($username);
480: 
481:         if ($u->id()) {
482:             $u->saveToSession();
483:             return true;
484:         } else {
485:             return false;
486:         }
487:     }
